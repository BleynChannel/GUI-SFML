5void GUI::BaseButton::handleEvent(const sf::Event& event)
6{
7 if (event.type == sf::Event::MouseButtonPressed &&
8 event.mouseButton.button == sf::Mouse::Button::Left &&
9 isPoint(static_cast<float>(event.mouseButton.x), static_cast<float>(event.mouseButton.y)))
10 {
11 pressed = !pressed;
12
13 Animation* tmpAnimation = (*animations)[0];
14 if (animatedPressed && tmpAnimation) {
15 tmpAnimation->stop();
16
17 tmpAnimation->start(animationPressedDuration);
18 }
19
20 isMousePressed = true;
21 } else if (isMousePressed &&
22 event.type == sf::Event::MouseButtonReleased &&
23 event.mouseButton.button == sf::Mouse::Button::Left)
24 {
25 Animation* tmpAnimation = (*animations)[0];
26
27 if (isPoint(static_cast<float>(event.mouseButton.x), static_cast<float>(event.mouseButton.y))) {
28 pressed = switchMode ? pressed : false;
29
30 if (animatedPressed && tmpAnimation) {
31 float tmpPosition = tmpAnimation->position;
32 tmpAnimation->stop();
33
34 tmpAnimation->start(tmpPosition * animationPressedDuration);
35 }
36
37 if (onTouch)
38 onTouch(this, event.mouseButton.x, event.mouseButton.y);
39 } else {
40 pressed = !pressed;
41
42 if (animatedPressed && tmpAnimation) {
43 float tmpPosition = tmpAnimation->position;
44 tmpAnimation->stop();
45
46 tmpAnimation->start(tmpPosition * animationPressedDuration);
47 }
48 }
49
50 isMousePressed = false;
51 } else if (hoverMode && !isMousePressed && event.type == sf::Event::MouseMoved) {
52 Animation* tmpAnimation = (*animations)[0];
53 bool stateAnimationPressed = false;
54 if (tmpAnimation)
55 stateAnimationPressed = tmpAnimation->started;
56 else
57 stateAnimationPressed = false;
58
59 if (!stateAnimationPressed) {
60 bool mouseInButton = isPoint(static_cast<float>(event.mouseMove.x), static_cast<float>(event.mouseMove.y));
61
62 if (!hover && mouseInButton) {
63 hover = true;
64
65 tmpAnimation = (*animations)[1];
66 if (animatedHover && tmpAnimation) {
67 tmpAnimation->stop();
68
69 tmpAnimation->start(animationPressedDuration);
70 }
71 } else if (hover && !mouseInButton) {
72 hover = false;
73
74 tmpAnimation = (*animations)[1];
75 if (animatedHover && tmpAnimation) {
76 float tmpPosition = tmpAnimation->position;
77 tmpAnimation->stop();
78
79 tmpAnimation->start(tmpPosition * animationPressedDuration);
80 }
81 }
82 }
83 }
