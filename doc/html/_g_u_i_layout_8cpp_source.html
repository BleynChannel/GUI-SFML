24void GUI::Layout::calculatePercentageSize(bool changingExistent, std::vector<Column>& array, uint32_t index)
25{
26 if (array[index].sizePolicy == SizePolicy::Relative) {
27 float allPercents = 0.f, beforeFromIndexPercents = 0.f;
28
29 uint32_t sizeArray = array.size();
30 Column* line;
31 for (uint32_t i = 0U; i < sizeArray; i++) {
32 line = &array[i];
33
34 allPercents += line->sizePolicy == SizePolicy::Relative ?
35 line->size :
36 0.f;
37
38 beforeFromIndexPercents += (line->sizePolicy == SizePolicy::Relative && i <= index) ?
39 line->size :
40 0.f;
41 }
42
43 if (changingExistent && beforeFromIndexPercents > 1.f)
44 throw Error("Layout summary percentage > 100% at index " + std::to_string(index), "Layout::calculatePercentageSize");
45
46 float modifyer = changingExistent ?
47 (1.f - beforeFromIndexPercents) / (allPercents - beforeFromIndexPercents) :
48 (1.f - array[index].size) / (allPercents - array[index].size);
49
50 for (uint32_t i = changingExistent ? (index + 1U) : 0U; i < sizeArray; i++)
51 array[i].size *= (array[i].sizePolicy == SizePolicy::Relative && i != index) ?
52 modifyer :
53 1.f;
54 }

544void GUI::Layout::update()
545{
546 float columnPercentageTotalSize = static_cast<float>(size.x - getAbsoluteTotalSize(*columns));
547 float rowPercentageTotalSize = static_cast<float>(size.y - getAbsoluteTotalSize(*rows));
548
549 Cell* cell;
550 uint32_t sizeChildrens = positionChildrens->size();
551
552 uint32_t sizeColumn = columns->size(), sizeRow = rows->size();
553 for (uint32_t row = 0U; row < sizeRow; row++) {
554 getCalculatedSize((*rows)[row], rowPercentageTotalSize);
555
556 for (uint32_t column = 0U; column < sizeColumn; column++) {
557 getCalculatedSize((*columns)[column], columnPercentageTotalSize);
558
559 cell = &(*grid)[row][column];
560
561 cell->countChildToCell = 0U;
562
563 for (auto& child : *positionChildrens)
564 if (child.column == column && child.row == row)
565 cell->countChildToCell++;
566
567 cell->childsWidth = 0U;
568 cell->childsHeight = 0U;
569 }
570 }
571
572 sf::Vector2u sizeChild;
573
574 for (auto& child : *positionChildrens)
575 if (!child.object->getHidden()) {
576 if (child.row >= 0U && child.row < sizeRow && child.column >= 0U && child.column < sizeColumn) {
577 cell = &(*grid)[child.row][child.column];
578 sizeChild = child.object->getSize();
579
580 sizeChild.x = cell->fitting.horizontal ?
581 (((*columns)[child.column].calculatedSize - cell->fitting.horizontalOffset) /
582 (cell->direction == Direction::Horizontal ?
583 cell->countChildToCell :
584 1U)) :
585 sizeChild.x;
586
587 sizeChild.y = cell->fitting.vertical ?
588 (((*rows)[child.row].calculatedSize - cell->fitting.verticalOffset) /
589 (cell->direction == Direction::Vertical ?
590 cell->countChildToCell :
591 1U)) :
592 sizeChild.y;
593
594 if (cell->fitting.horizontal || cell->fitting.vertical)
595 child.object->setSize(sizeChild);
596
597 cell->childsWidth = cell->direction == Direction::Horizontal ?
598 (cell->childsWidth + sizeChild.x + cell->spacing) :
599 (cell->childsWidth > sizeChild.x ? cell->childsWidth : sizeChild.x);
600
601 cell->childsHeight = cell->direction == Direction::Vertical ?
602 (cell->childsHeight + sizeChild.y + cell->spacing) :
603 (cell->childsHeight > sizeChild.y ? cell->childsHeight : sizeChild.y);
604 } else {
605 child.object->setDisabled(true);
606 child.object->setHidden(true);
607 }
608 }
609
610 float x = 0.f, y = 0.f;
611 uint32_t calcWidth, calcHeight;
612
613 for (uint32_t row = 0U; row < sizeRow; row++) {
614 for (uint32_t column = 0U; column < sizeColumn; column++) {
615 cell = &(*grid)[row][column];
616 calcWidth = (*columns)[column].calculatedSize;
617 calcHeight = (*rows)[row].calculatedSize;
618
619 cell->pos = cell->alignment.getCoordinates(
620 x, y, calcWidth, calcHeight,
621 cell->childsWidth - (cell->direction == Direction::Horizontal ? cell->spacing : 0U),
622 cell->childsHeight - (cell->direction == Direction::Vertical ? cell->spacing : 0U)
623 );
624
625 if (cell->margin.horizontal || cell->margin.vertical)
626 cell->pos = cell->margin.getCoordinates(
627 cell->pos,
628 cell->alignment.horizontal, cell->alignment.vertical
629 );
630
631 x += static_cast<float>(calcWidth);
632 }
633
634 x = 1.f, y += static_cast<float>(calcHeight);
635 }
636
637 sf::Vector2f alignmentPos;
638 float newChildX, newChildY;
639
640 for (auto& child : *positionChildrens) {
641 cell = &(*grid)[child.row][child.column];
642 sizeChild = child.object->getSize();
643
644 alignmentPos = cell->alignment.getCoordinates(
645 cell->pos.x, cell->pos.y,
646 cell->childsWidth, cell->childsHeight,
647 sizeChild.x, sizeChild.y
648 );
649
650 newChildX = cell->direction == Direction::Horizontal ?
651 cell->pos.x :
652 alignmentPos.x;
653
654 newChildY = cell->direction == Direction::Vertical ?
655 cell->pos.y :
656 alignmentPos.y;
657
658 cell->pos.x += cell->direction == Direction::Horizontal ?
659 (sizeChild.x + cell->spacing) :
660 0.f;
661
662 cell->pos.y += cell->direction == Direction::Vertical ?
663 (sizeChild.y + cell->spacing) :
664 0.f;
665
666 child.object->setPosition(newChildX, newChildY);
667 }
668
669 updateLayout = false;
670}
